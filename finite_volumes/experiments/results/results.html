def result_plot(model, xlim=None):
    rs = np.concatenate([np.linspace(0, model.xs[0], 6), model.xs[1:]])
    phi = np.concatenate([model.additional[-1, -10:-3].to_numpy(), model.phi[-1, 1:].to_numpy()])
    plt.figure(figsize=(10, 5))
    plt.grid(True)
    if xlim is not None:
        plt.xlim(xlim)
    plt.xlabel('$r$')
    plt.ylabel(r'$\phi$')
    plt.plot(rs, phi)
    plt.show()
